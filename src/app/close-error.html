Closing Error Modal...

If the error modal doesn't close automatically, copy and paste this into your browser console:

// Close error modal
document.querySelectorAll('[role="dialog"], .error-overlay, .modal-overlay').forEach(el => el.remove());
document.querySelectorAll('body > div').forEach(div => { if (div.style.position === 'fixed') div.remove(); });
// Navigate to admin
window.location.href = '/admin';
